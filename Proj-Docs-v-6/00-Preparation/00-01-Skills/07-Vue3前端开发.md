### 7.2 Axios 封装（含多租户支持）

```typescript
// src/api/request.ts
import axios from 'axios'
import { ElMessage } from 'element-plus'
import { useUserStore } from '@/stores/user'

const request = axios.create({
  baseURL: '/admin-api',
  timeout: 30000
})

// 请求拦截器：注入 Token 和租户ID
request.interceptors.request.use(
  (config) => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`
    }
    // 多租户：注入租户ID
    if (userStore.tenantId) {
      config.headers['tenant-id'] = userStore.tenantId
    }
    return config
  },
  (error) => Promise.reject(error)
)

// 响应拦截器：统一处理 code/message
request.interceptors.response.use(
  (response) => {
    const { code, msg, data } = response.data
    if (code === 0) {
      return data
    }
    ElMessage.error(msg || '请求失败')
    return Promise.reject(new Error(msg))
  },
  (error) => {
    if (error.response?.status === 401) {
      ElMessage.error('登录已过期，请重新登录')
      window.location.href = '/login'
    } else {
      ElMessage.error('网络错误，请稍后重试')
    }
    return Promise.reject(error)
  }
)

export default request
```

### 7.3 API 层

```typescript
// src/api/system/user/index.ts
import request from '@/api/request'

export interface UserVO {
  id: number
  username: string
  nickname: string
  email: string
  mobile: string
  status: number
  deptId: number
  createTime: string
}

export interface UserPageReqVO {
  pageNo: number
  pageSize: number
  username?: string
  mobile?: string
  status?: number
}

export interface UserCreateReqVO {
  username: string
  password: string
  nickname: string
  email: string
  mobile: string
  status: number
  deptId: number
  roleIds: number[]
}

export interface UserUpdateReqVO {
  id: number
  nickname: string
  email: string
  mobile: string
  status: number
  deptId: number
  roleIds: number[]
}

export interface PageResult<T> {
  records: T[]
  total: number
  pageNo: number
  pageSize: number
}

// 分页查询
export function getUserPage(params: UserPageReqVO) {
  return request.get<any, PageResult<UserVO>>('/system/user/page', { params })
}

// 查询详情
export function getUser(id: number) {
  return request.get<any, UserVO>('/system/user/get', { params: { id } })
}

// 新增
export function createUser(data: UserCreateReqVO) {
  return request.post<any, number>('/system/user/create', data)
}

// 更新
export function updateUser(data: UserUpdateReqVO) {
  return request.put<any, boolean>('/system/user/update', data)
}

// 删除
export function deleteUser(id: number) {
  return request.delete<any, boolean>('/system/user/delete', { params: { id } })
}
```

const loading = ref(false)

const dialogVisible = ref(false)
const isEdit = ref(false)
const formRef = ref()
const submitting = ref(false)

  username: '', password: '', nickname: '', email: '', mobile: '', status: 0, deptId: 0, roleIds: []

  loading.value = true

    const res = await getUserPage(queryParams)
    tableData.value = res.records

  try {
    if (isEdit.value) {
      await updateUser(formData as UserUpdateReqVO)
      ElMessage.success('更新成功')

      await createUser(formData as UserCreateReqVO)
      ElMessage.success('创建成功')

    dialogVisible.value = false
    fetchData()

}

</script>

### 7.5 动态路由与权限控制

密讯ETM前端采用**后端驱动动态路由**模式：

```
用户登录
  │
  ▼
后端返回权限信息 (/admin-api/system/auth/get-permission-info)
  │
  ├── menus[]  → 动态渲染侧边栏菜单
  └── permissions[] → 控制按钮显示 (v-permission)
  │
  ▼
前端根据 menus 动态注册路由
  │
  ▼
渲染完整页面
```

```typescript
// 路由守卫简例
router.beforeEach(async (to, _from, next) => {
  const userStore = useUserStore()
  if (userStore.token) {
    if (!userStore.permissions.length) {
      // 首次进入，获取权限信息
      await userStore.getPermissionInfo()
      // 动态生成路由
      const routes = generateRoutes(userStore.menus)
      routes.forEach(r => router.addRoute(r))
      next({ ...to, replace: true })
    } else {
      next()
    }
  } else {
    to.path === '/login' ? next() : next('/login')
  }
})
```

## 常见问题

**Q：Vue 3 的 `<script setup>` 和 Options API 有什么区别？**

`<script setup>` 是 Composition API 的语法糖，代码更简洁，不需要 return 模板变量。密讯ETM项目统一使用 `<script setup lang="ts">`。

**Q：动态路由添加后刷新页面空白？**

检查路由守卫逻辑——刷新时 `userStore.token` 和 `userStore.permissions` 的恢复时机。通常需要将 token 持久化到 localStorage，并在路由守卫中 await 权限恢复完成。

**Q：Vite 开发时代理后端接口 404？**

检查 `vite.config.ts` 中的 proxy 配置，确保 `/admin-api` 和 `/app-api` 都正确代理到后端地址。
